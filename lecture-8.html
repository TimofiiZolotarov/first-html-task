void Simplify(Fraction& f)
{
    if (f.num == 0) { f.den = 1; return; }
    int a = abs(f.num), b = f.den;
    while (a != b)
        if (a > b) a -= b; else b -= a;
    f.num /= a;
    f.den /= a;
}
